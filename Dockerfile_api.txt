# base image
FROM python:3.10.1

# set working directory
WORKDIR /app

# copy dependency files
COPY requirements.txt .

# install dependency
RUN pip install --no-cache-dir -r requirements.txt

# copy
COPY . .

# expose
EXPOSE 8000

# command
CMD ["uvicorn", "api:app", "--host", "0.0.0.0", "--port", "8000"]
